    public static Vector union(Vector a, Vector b) {

            System.out.print("Vector a is Null");
        } else if (b.isEmpty()) {
            System.out.print("Vector b is Null");
        } else {
            Vector c = new Vector();
            c.addAll(a);
            c.addAll(b);
            return c;
        }
    }

    public static void main(String[] args) {
        Vector a = new Vector();
        a.add(4);
        Vector b = new Vector();
        b.add(5);
        Vector c = union(a, b);
